#include <stdio.h>
#include <stdlib.h>




int main()
{
//declaring the variable
   float number, counter=0, sum=0, average;


//stopping condition
   if(number<0) {
    printf("\n ERROR, please try again: \n");
    scanf("%f", &number);
   }

//interactive input
   while(number!=0) {
    printf("\n Hello User, please insert one POSITIVE integer: \n");
    scanf("%f", &number);
    if(number<0) {
            printf("\n ERROR, please try again: \n");
    scanf("%f", &number);

    }
    sum+=number;
    counter++;
   }

//calculating the average number
   average=sum/counter;
   if(average!=0) {
   printf("Average=%f\n", sum/((float)counter-1));;
   } else {
   printf("No average number.");
   }



    return 0;
     }



//endmain
