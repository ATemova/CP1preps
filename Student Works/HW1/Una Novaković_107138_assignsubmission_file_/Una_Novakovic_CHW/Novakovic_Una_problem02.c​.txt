#include <stdio.h>
#include <stdlib.h>



int main()
{
    //declaring the variables
    int N;

    //interactive input
    printf(" Please, insert one integer: \n");
    scanf("%d", &N);

    for(int i=0;i<N;i++) {//for loop to go through the rows
        for(int j=0;j<N;j++) {//for loop to print *
                if(j<i) {
                    printf(" ");
                } else {
                printf("*");

                }

        }
        printf("\n");
    }



return 0;
}//endmain
