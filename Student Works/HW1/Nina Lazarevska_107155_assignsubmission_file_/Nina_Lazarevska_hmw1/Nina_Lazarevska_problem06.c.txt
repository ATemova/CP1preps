#include<stdio.h>
int main(){
    //declaring variables
    int a[30], i,  smallest;

    //telling the user what to type in
    printf("\nEnter 10 elements :");
    //Reading the elements and putting them in an array
    for (i = 0; i <= 10; i++){
        scanf("%d", &a[i]);
    }//end for
    //Assume that the first element is the smallest
    smallest = a[0];

    //comparing every number in the array and every time a number is smaller then
    //the last number that was memorized in the "smallest" we swap its places
    //thats how we find the smallest
    for (i = 0; i <= 10; i++) {
      if (a[i] < smallest) {
         smallest = a[i];
        }
    }

    // Printing the Result
    printf("\nSmallest Element : %d", smallest);

return (0);
}//end main
