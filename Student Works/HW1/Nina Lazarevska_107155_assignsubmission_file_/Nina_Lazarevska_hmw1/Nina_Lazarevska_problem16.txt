/*Problem 16.c
By listing the first six prime numbers: 2, 3, 5, 7, 11, and 13, we can see that the 6th prime is 13.
The first six prime numbers are 2, 3, 5, 7, 11, and 13. Write a C programming to compute for the
1001st prime number?*/

#include <stdio.h>
#include<math.h>
int main ()
{
  int nth, c = 0, num = 2, i, primeNum = 0;
  printf ("Which prime number do you want?\n");
  scanf ("%d", &nth);

  while (c != nth)
    {
      int count = 0;
      for (i = 2; i <= sqrt (num); i++)
 {
   if (num % i == 0)
     {
       count++;
       break;
     }
 }
      if (count == 0)
 {
   c++;
   primeNum = num;
 }
      num = num + 1;
    }
  printf ("%dth prime number is %d ",nth,primeNum);

}//end main
