/*Problem 13.c
Write a C program to input elements in an array. After all the numbers are inserted in the array,
put even and odd elements in two separate arrays.
*/

#include<stdio.h>
int main(){

//declaring variables
    int i, j = 0, k = 0, n;
//getting the size of the array
    printf("How many numbers do you want to enter:\n");
    scanf("%d", &n);
    //declaring an array for all the numbers, an array for the odd numbers and an array for the even numbers.
    long int arr[n], odd[n], even[n];
    printf("Now enter the numbers:\n");
    for (i = 0; i < n; i++){
            scanf("%ld", &arr[i]);
            fflush(stdin);
        }

        //Putting odd numbers in the array odd and even elements into the array even. */

        for (i = 0; i < n; i++){
            if (arr[i] % 2 == 0){
                even[j] = arr[i];
                j++;
            }//end if
            else{
                odd[k] = arr[i];
                k++;
            }//end else
        }
//printing the odd numbers
    printf("Odd numbers are:\n");
    for (i = 0; i < k; i++){
            printf("%ld, ", odd[i]);
        }
    printf("\n");
//printing the even numbers
    printf("Even numbers are:\n");
    for (i = 0; i < j; i++){
            printf("%ld, ", even[i]);
        }
return 0;
}//end main
