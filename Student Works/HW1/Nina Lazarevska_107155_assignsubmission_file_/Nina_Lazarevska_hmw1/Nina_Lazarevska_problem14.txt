/*Problem 14.c
Write a program in C to count a total number of duplicate elements in an array. Minimum should
be 5 and maximum is 25 elements. Inputs must be integers (both negative and positive).
*/
#include<stdio.h>
int main(){
//declaring variables and array
    int arr[20];
    int i, j, size, count = 0;

    // Getting the size of the array from the user
    printf("Enter size of the array : ");
    scanf("%d", &size);
    if(size<5){
        printf("You have to enter between more then 5, but less then 25 elements!!!");
    }
    else if(size>25){
         printf("You have to enter between more then 5, but less then 25 elements!!!");
    }
    else{

    // Input elements in array
    printf("Enter elements in array : ");
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }


    //Find all duplicate elements in array

    for(i=0; i<size; i++)
    {
        for(j=i+1; j<size; j++)
        {
            //If a duplicate number found then increment the variable count by 1
            if(arr[i] == arr[j])
            {
                count++;
                break;
            }
        }
    }

    printf("\nThe number of duplicate numbers is: %d", count);
    }
    return 0;

}//end main
