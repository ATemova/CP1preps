/*Write a C program that will perform the following functions below:
Allow the user to enter a number. This number will determine the maximum number of asterisks
in a row. Print a left-based triangle that will have the userâ€™s input number as the maximum
number. For example, if the user inputs the number 5, the output should be as displayed below:
*
**
***
****
*****
****
***
**
*
*/
#include<stdio.h>
int main(){
//declaring variables
int n;

//getting input
printf("Enter a number:");
scanf("%d", &n);

//the result
printf("Here is the left-based triangle:");
for(int i=1; i<=n; i++){
    for(int j=1; j<=i; j++){
            printf("*");
    }
    printf("\n");
}

for(int i=n-1; i>=0; i--){
    for(int j=1; j<=i; j++){
            printf("*");
    }
    printf("\n");
}


return 0;
}//end main
