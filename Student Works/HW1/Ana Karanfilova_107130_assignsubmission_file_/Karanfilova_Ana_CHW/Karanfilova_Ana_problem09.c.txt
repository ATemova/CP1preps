/*Calendar Generator: A program that will show the specific month. It will ask the user which day
of the week it will start, the date and how many days does it have.*/
#include<stdio.h>
#include<string.h>

int main(){
    //declaring variables
    int dDays, dStart, dToday=1;
    //initializing the days of the week
    char sWeek[]="Sun\tMon\tTue\tWed\tThu\tFri\tSat";
    //interactive input
    printf("Which day of the week does the month start?\n");
    printf("0: Sun, 1: Mon, 2: Tue, 3: Wed, 4: Thu, 5: Fri, 6: Sat\n");
    scanf("%d", &dStart);
    printf("\nHow many days does the month have?\n");
    scanf("%d", &dDays);
    //printing the calendar
    printf("\n");
    puts(sWeek);//printing days of the week
    //loop for weeks
    for(int dCtr1=1;dCtr1<=5;dCtr1++){
        //loop for days
        for(int dCtr2=1;dCtr2<=7;dCtr2++){
            if(dCtr2<=dStart && dCtr1==1){//starting from day of input
                printf("\t");
            }//end if
            else if(dToday>dDays){//stopping after specified input
                break;
            }//end else if
            else{
                printf("%d\t", dToday);//printing day by day
                dToday++;
            }//end else
        }//end nested for
        printf("\n");//marks new week
    }//end outer for
    return 0;
}//end main
