#include <stdio.h>
#define size 50
int main()
{
    int T,i,N;
    N=2;
    int arr[size];
    printf("Enter the total number of test case 1<=T<=200000 \t");
    scanf("%d",&T);
    printf("Enter the value of N 1<=N<=200000 \n");
    for(i=1;i<=T;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Output are\n");
    for(i=1;i<=T;i++)
    {
        printf("%d\n",arr[i]*N);
    }
    return 0;
}
